	import pandas as pd
	
	from DNACauth import DNACauth
	
	#Auth -
	base_url = "https://prod-apl-dnac/dna"
	site_endpoint = "/intent/api/v1/topology/site-topology"
	req_headers = DNACauth()
	
	
	#Start Request ------------------
	site_endpoint = "/intent/api/v1/network-device"
	
	devices = requests.get(url=f"{base_url}{site_endpoint}",headers=req_headers,verify=False).json()['response']
	
	print(json.dumps(devices, indent=2))
	
	for device in devices:
		if device['hostname'] == "###hostname###":
			dev_id = device['id']
	
	
	dev_vlan_endpoint = f"/intent/api/v1/network-device/{dev_id}/vlan"
	
	dev_vlans = requests.get(url=f"{base_url}{dev_vlan_endpoint}",headers=req_headers,verify=False).json()['response']
	
	print(json.dumps(dev_vlans, indent=2))
	
	text = json.dumps(dev_vlans)
	
	df = pd.read_json(text)
